func areDivisors(number num1, number num2)
    return ((num1 % num2 = 0) or (num2 % num1 = 0))

func main()
    begin
        var num1 <- readNumber() 
        var num2 <- readNumber()

        if (areDivisors(num1, num2)) writeString("Yes")
        else writeString("No")
    end

func isPrime(number x)
    begin
        if (x <= 1) return false 
        var i <- 2
        for i until i > x / 2 by 1 
        begin
            if (x % i = 0) return false
        end
        return true
    end

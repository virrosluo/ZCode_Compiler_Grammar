func sum(number arr[10]) begin
            var total <- 0
            for i until len(arr) by 1 begin
                total <- total + arr[i]
            end
            return total
        end


        func main() begin
            number numbers[5] <- [1,2,3,4,5]
            var result <- sum(numbers)
            print("The sum of the numbers is: ",result)
        end
